    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const menuButton = document.getElementById('mobile-menu-button');
            const mobileMenu = document.getElementById('mobile-menu');
            const navLinks = mobileMenu.querySelectorAll('a');
            const form = document.querySelector('form');
            const formMessage = document.getElementById('form-message');
            const errorMessage = document.getElementById('error-message');

            // --- Mobile Menu Toggle ---
            menuButton.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });

            // Close mobile menu when a link is clicked
            navLinks.forEach(link => {
                link.addEventListener('click', () => {
                    mobileMenu.classList.add('hidden');
                });
            });

            // --- Dynamic Footer Year ---
            document.getElementById('current-year').textContent = new Date().getFullYear();

            // --- Form Submission Simulation (No real backend) ---
            form.addEventListener('submit', (e) => {
                e.preventDefault();
                form.querySelector('button[type="submit"]').textContent = 'Sending...';
                formMessage.classList.add('hidden');
                errorMessage.classList.add('hidden');
                
                // Collect form data (for potential Firestore implementation later)
                const formData = {
                    name: document.getElementById('name').value,
                    email: document.getElementById('email').value,
                    service: document.getElementById('service').value,
                    message: document.getElementById('message').value,
                    timestamp: new Date().toISOString()
                };
                console.log("Simulating form submission with data:", formData);

                // Simulate network delay
                setTimeout(() => {
                    try {
                        // Simulate successful submission
                        form.reset();
                        formMessage.classList.remove('hidden');
                        form.querySelector('button[type="submit"]').textContent = 'Send General Inquiry';
                        
                        // Hide success message after 5 seconds
                        setTimeout(() => {
                            formMessage.classList.add('hidden');
                        }, 5000);

                    } catch (error) {
                        // In a real app, this catches fetch/network errors
                        errorMessage.classList.remove('hidden');
                        form.querySelector('button[type="submit"]').textContent = 'Send General Inquiry';
                        console.error("Form submission failed:", error);
                    }
                }, 1500); // 1.5 second delay
            });
        });
    </script>
